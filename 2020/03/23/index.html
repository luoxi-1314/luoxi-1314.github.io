MySQL

选择语句

select * 
from customers
-- where customers_id = 1
-- order by first_name       
-- "-- "相当于注释
select 
    last_name,
    (points + 10) * 10 as discount_factor
from customers
-- 1、select可以使用运算符；
-- 2、as起别名,若要加入空格可用 ' '

select distinct state
FROM customers;
-- distance 查询数据并去掉重复项

WHERE子句

select *
FROM customers
where points > 3000
-- where中可以使用算数运算、逻辑运算
select *
from customers
where state in ('VA','FL')
-- in的用法
select *
from customers
where points between 100 and 300
-- between的用法
select *
from customers
where last_name like 'b%'
-- like的用法
-- %多个字符占位符
-- _单个字符占位符

正则表达式

select *
from customers
where last_name regexp 'field$'
元字符 描述
^ 开始
& 结尾
| 或
[] 包含
， 分割 
select *
from customers
where phone is null 
-- null的用法
select *
from customers
order by first_name desc
-- order by 用法：用于排序 
-- desc 降序排列
-- asc 升序
select *
from customers
limit 5,3
--limit用法：根据位置查数据

数据库连接

-- 内连接
select *
from orders
join customers
    on orders.customer_id = customers.customer_id 
-- join 多张表里查询 
-- 跨数据库连接
use sql_inventory;
select *
from store.order_items oi
join sql_inventory.products p
    on oi.product_id = p.product_id
-- 自连接
use sql_hr;
select 
    e.employee_id,
    e.first_name,
    m.first_name as manager
from employees e
join employees m
    on e.reports_to = m.employee_id
-- 多表连接
use store;
select *
from orders o
join customers c
    on o.customer_id = c.customer_id
join order_statuses os
    on o.status = os.order_status_id
-- 复合条件连接
select *
from order_items oi
join order_item_notes oin
    on oi.order_id = oin.order_id
    and oi.product_id = oin.product_id
-- 隐含连接
select *
from orders o, customers c
where o.customer_id = c.customer_id
-- 外连接
select 
    c.customer_id,
    c.first_name,
    o.order_id
from customers c
join orders o
    on c.customer_id = o.customer_id
order by c.customer_id
-- 可以加left right
-- left join ...

-- USING子句
select 
    o.order_id,
    o.order_date,
    c.first_name as customer
from orders o
join customers c
    on o.customer_id  = c.customer_id
join shippers sd
    on o.shipper_id = sh.shipper_idcustomers
-- USING 可以简洁ON语法
-- 自然连接
select 
    o.order_id,
    c.first_name
from orders o
natural join customers c
-- 交叉连接 ： 笛卡尔积
select 
    c.first_name as customer,
    p.name as product
from customers c
cross join products p
order by c.first_name
-- 联合
select first_name
from customers
union 
select name
from shippers
-- 将两个sql语句联合在一起

增删改

-- 插入单行
insert into customers (
    first_name,
    last_name,
    birth_date,
    address,
    city,
    state)
values ( 
    'John' ,
    'Smith',
    '1900-01-01',
    'address',
    'city',
    'CA')
-- 插入多行
insert into shippers (name)
values  ('shipper1'), 
        ('shipper2')
-- 创建表复制
create table orders_archived as 
select * from orders
-- 更新单行
update invoices
set payment_total = 10, payment_date = '2019-03-01'
where invoice_id = 1
-- 更新多行
update customers
set points = points + 50
where birth_date < '1990-01-01'
-- update中用子句查询
update invoices
set 
    payment_total = invoice_total * 50,
    payment_date = due_date
where client_id in
                (select client_id
                from clients
                where state in ('CA','NY'))
-- 删除行
delete from invoices
where invoice_id = 1
-- 聚合函数
select 
    max(invoice_total) as highest,
    min(invoice_total) as lowest,
    avg(invoice_total) as average,
    sum(invoice_total) as total_of_invoices,
    count(payment_date) as count_of_number
from invoices
-- group by分组查看
select
    client_id,
    sum(invoice_total) as total_sales
from invoices
group by client_id
-- having 筛选数据
select
    client_id,
    sum(invoice_total) as total_sales
from invoices
group by client_id
having total_sales > 500
-- rollup 分组汇总
select
    state,
    city,
    sum(invoice_total) as total_sales
from invoices i
join clients c using (client_id)
group by client_id with rollup
